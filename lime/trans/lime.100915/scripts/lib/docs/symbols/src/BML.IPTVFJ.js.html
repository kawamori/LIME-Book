127    *  @param   {Date}   base 基準となるDateオブジェクト
128    *  @param   {Number} time 加算する時間
129    *  @param   {Number} unit timeの単位(0:ms/1:sec/2:min/3:hour/4:day/5:week)
130    *  @returns {Date}        加算した結果のDateオブジェクト(失敗した場合はbase)

132   function addDate(base, time, unit) {
133     var t = base.getTime();
134     if      (unit === 0) t += time;
135     else if (unit ==  1) t += time * 1000;
136     else if (unit ==  2) t += time * 1000 * 60;
137     else if (unit ==  3) t += time * 1000 * 60 * 60;
138     else if (unit ==  4) t += time * 1000 * 60 * 60 * 24;
139     else if (unit ==  5) t += time * 1000 * 60 * 60 * 24 * 7;
140     var ret = new Date; ret.setTime(t);
141     return(ret);
142   }

160   /** 指定した数値の3桁ごとに","を挿入した文字列を求める．

163    *  @returns {String}       数値を文字列化した結果(失敗した場合はnull)

165   function formatNumber(value) {
166     value = String(value);
167     var len = value.length;
168     if (len < 3) return(value);
169     
170     var odd = len % 3, buf = [];
171     if (odd > 0) buf.push(value.substring(0, odd));

174     }
